# -----------------------------------------------------------------------------
# FastAPI 백엔드 서버 실행 패키지

# FastAPI 웹 API 서버를 만들기 위한 핵심 프레임워크입니다.

# - docs/free-deployment-guide.md: Render에 FastAPI 백엔드 배포
fastapi==0.115.6

# FastAPI 앱을 실제 서버로 실행하는 ASGI 서버입니다.
# 로컬에서는 uvicorn main:app --reload 명령으로 실행하고,
# Render 배포에서는 uvicorn main:app --host 0.0.0.0 --port $PORT 형태로 실행합니다.
# 사용 위치:
# - 로컬 백엔드 실행
# - Render 백엔드 배포
uvicorn==0.32.1


# -----------------------------------------------------------------------------
# 요청/응답 데이터 검증 패키지
# -----------------------------------------------------------------------------

# API로 들어오는 요청 Body와 API가 반환하는 응답 구조를 검증하는 패키지입니다.
# 이 프로젝트에서는 ChatRequest, ChatResponse 같은 모델을 정의할 때 사용합니다.

# - /api/chat: 빈 질문이나 잘못된 요청을 막는 기본 검증

# -----------------------------------------------------------------------------
# 환경 변수 관리 패키지

# - docs/free-deployment-guide.md: Render 환경변수와 로컬 .env의 차이 설명
python-dotenv==1.0.1
