# 05_llm-agent-mini-project 공통 설치 파일입니다.
#
# 기본 흐름은 05 과정 최상위 폴더에서 .venv 하나를 만들고,
# 이 파일로 프로젝트에 필요한 패키지를 한 번에 설치하는 방식입니다.
#
# 사용 예시:
#   cd C:\aidev\05_llm-agent-mini-project
#   python -m venv .venv
#   .\.venv\Scripts\Activate.ps1
#   python -m pip install --upgrade pip
#   pip install -r requirements.txt

# OpenAI API를 호출할 때 사용합니다.
# 사용 위치:
# - 02_instructor-sample-project
# - 99_team-projects/llm-agent-team-template
openai

# .env 파일에서 OPENAI_API_KEY, OPENAI_MODEL 같은 환경 변수를 읽을 때 사용합니다.
# API Key를 코드에 직접 쓰지 않기 위한 필수 보조 패키지입니다.
python-dotenv

# 외부 API를 Tool처럼 연결하거나, 간단한 HTTP 요청 예제를 만들 때 사용합니다.
# 실제 Calendar API 등을 붙일 때 확장할 수 있습니다.
httpx

# Agent State, Tool 입력값, Tool 출력값 같은 데이터 구조를 검증할 때 사용합니다.
# 프로젝트 문서의 타입 설계와 코드 구현을 연결하는 데 도움이 됩니다.
pydantic

# Agent 실행 흐름을 State, Node, Edge 기반 그래프로 구성할 때 사용합니다.
# 05 프로젝트의 핵심 패키지입니다.
langgraph

# Prompt, Model, Parser, Tool 연결 구조를 이해하고 필요한 경우 프로젝트에 적용할 때 사용합니다.
# 05에서는 필수 개념을 다시 깊게 배우기보다 04에서 배운 내용을 프로젝트에 적용합니다.
langchain

# LangChain/LangGraph 예제에서 OpenAI 모델을 연결할 때 사용합니다.
langchain-openai

# LangChain/LangGraph의 공통 Runnable, 메시지, 출력 처리 구조를 사용할 때 필요합니다.
langchain-core

# Agent 실행 결과를 간단한 웹 화면으로 보여줄 때 사용합니다.
# 최종 프로젝트 UI는 Streamlit으로 구성합니다.
streamlit

# 선택 확장 패키지입니다.
# 팀 프로젝트에서 로컬 pgvector PostgreSQL에 RAG 또는 장기 기억 데이터를 저장할 때 사용합니다.
# PostgreSQL 서버 자체는 pip로 설치되지 않습니다.
# 필요한 경우 SETUP.md의 docker run 명령으로 pgvector 컨테이너를 실행합니다.
psycopg[binary]
