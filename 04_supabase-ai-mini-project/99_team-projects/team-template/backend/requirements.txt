# 이 파일은 팀 프로젝트 백엔드를 Render 같은 서비스에 따로 배포할 때 사용합니다.
# Render의 Root Directory를 backend로 지정하면 이 파일을 기준으로 패키지가 설치됩니다.

# FastAPI API 서버를 만들기 위한 핵심 웹 프레임워크입니다.
# 팀 프로젝트에서는 service_logs 저장/조회/수정 API와 상태 확인 API를 만듭니다.
fastapi==0.115.6

# FastAPI 앱을 실행하는 ASGI 서버입니다.
# 로컬 실행:
#   uvicorn main:app --reload --host 127.0.0.1 --port 8000
# Render 배포:
#   uvicorn main:app --host 0.0.0.0 --port $PORT
uvicorn[standard]==0.32.1

# 04_supabase-ai-mini-project 최상위 .env 파일을 읽기 위해 사용합니다.
# 배포 환경에서는 Render Environment Variables에 등록한 값을 읽습니다.
python-dotenv==1.0.1

# Python에서 Supabase 테이블에 insert/select/update 요청을 보내기 위한 패키지입니다.
# 팀 프로젝트에서는 service_logs, messages, feedbacks 같은 테이블과 연결됩니다.
supabase==2.10.0

# FastAPI 요청 Body와 응답 구조를 검증하기 위한 패키지입니다.
# ServiceLogCreate, ServiceLogUpdate 같은 모델을 만들 때 사용합니다.
pydantic==2.10.4

# Gemini API를 Python SDK 방식으로 호출하기 위한 Google Gen AI SDK입니다.
# 팀 프로젝트에서 사용자 질문에 대한 AI 응답 생성 기능을 백엔드에 추가할 때 사용합니다.
# 03 과정의 기본 AI API는 Gemini이며, 모델 예시는 gemini-2.5-flash-lite입니다.
google-genai==1.2.0
