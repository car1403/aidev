# FastAPI는 backend API 서버를 만들 때 사용합니다.
# 07에서는 장애 이벤트 접수, Health Check, 복구 결과 조회 API를 구성할 때 사용합니다.
fastapi

# uvicorn은 FastAPI 애플리케이션을 실행하는 ASGI 서버입니다.
# Docker 컨테이너 안에서 backend 서비스를 실행할 때 사용합니다.
uvicorn

# Streamlit은 간단한 웹 화면을 빠르게 만들 때 사용합니다.
# frontend 화면과 monitor 대시보드 예제를 실행할 때 사용합니다.
streamlit

# httpx는 Python에서 HTTP 요청을 보낼 때 사용합니다.
# frontend/monitor/worker가 backend API를 호출하는 예제에서 사용합니다.
httpx

# pydantic은 요청/응답 데이터 구조를 명확하게 정의하고 검증할 때 사용합니다.
# 장애 이벤트, 복구 결과, Agent Context 같은 데이터를 다룰 때 사용합니다.
pydantic

# python-dotenv는 .env 파일의 환경변수를 Python 코드에서 읽을 때 사용합니다.
# API Key, 모델명, backend URL, 포트 번호 등을 코드 밖에서 관리할 수 있게 합니다.
python-dotenv

# openai는 OpenAI API를 호출할 때 사용합니다.
# 07에서는 필수 구현보다 선택 확장 항목에 가깝지만, Agent 판단 또는 결과 요약 예제로 사용할 수 있습니다.
openai

# langgraph는 Agent의 상태 전이 흐름을 그래프 형태로 구성할 때 사용합니다.
# 07에서 배운 Agent 흐름을 프로젝트 구조로 확장할 때 참고합니다.
langgraph

# langchain은 LLM, Prompt, Tool 연결 흐름을 구조화할 때 사용합니다.
# 08에서는 필수라기보다 05~07에서 만든 Agent 구조를 이해하고 확장할 때 사용합니다.
langchain

# langchain-openai는 LangChain에서 OpenAI 모델을 연결할 때 사용합니다.
# OpenAI 기반 Agent 예제를 확장할 때 사용합니다.
langchain-openai
